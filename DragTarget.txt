With Draggable, you can drag a widget anywhere, with DragTarget, you drop a widget on a target.

DragTarget is usually used together with Draggable.
The idea is:
    Draggable → "I can be dragged."
    DragTarget → "I can receive dropped items."

---------------------------------------------------------------------

import 'package:flutter/material.dart';

class DemoPage extends StatefulWidget {
  const DemoPage({super.key});

  @override
  State<DemoPage> createState() => _DemoPageState();
}

class _DemoPageState extends State<DemoPage> {
  @override
  Widget build(BuildContext context) {
    return const Scaffold(
      body: SafeArea(
        child: DragTargetWidget(),
      ),
    );
  }
}

class DragTargetWidget extends StatefulWidget {
  const DragTargetWidget({super.key});

  @override
  State<DragTargetWidget> createState() => _DragTargetWidgetState();
}

class _DragTargetWidgetState extends State<DragTargetWidget> {
  String droppedItem = 'Nothing';

  @override
  Widget build(BuildContext context) {
    return Padding(
      padding: const EdgeInsets.all(20),
      child: Column(
        mainAxisAlignment: MainAxisAlignment.center,
        spacing: 20,
        children: [
          // ==========================
          // DRAGGABLE ITEMS
          // ==========================
          Row(
            mainAxisAlignment: MainAxisAlignment.spaceEvenly,
            children: [
              Draggable<String>(
                data: 'Apple',

                feedback: _buildDragBox(
                  color: Colors.red,
                  label: 'Apple',
                ),

                childWhenDragging: Opacity(
                  opacity: 0.3,
                  child: _buildDragBox(
                    color: Colors.red,
                    label: 'Apple',
                  ),
                ),



                child: _buildDragBox(
                  color: Colors.red,
                  label: 'Apple',
                ),
              ),

              Draggable<String>(
                data: 'Banana',

                feedback: _buildDragBox(
                  color: Colors.yellow,
                  label: 'Banana',
                ),

                childWhenDragging: Opacity(
                  opacity: 0.3,
                  child: _buildDragBox(
                    color: Colors.yellow,
                    label: 'Banana',
                  ),
                ),

                child: _buildDragBox(
                  color: Colors.yellow,
                  label: 'Banana',
                ),
              ),
            ],
          ),

          // ==========================
          // DRAG TARGET
          // ==========================
          DragTarget<String>(
            onWillAcceptWithDetails: (details) {
              debugPrint('About to accept: ${details.data}');
              return true;
            },

            onAcceptWithDetails: (details) {
              setState(() {
                droppedItem = details.data;
              });
            },

            onLeave: (data) {
              debugPrint('$data left the target');
            },

            builder: (
                BuildContext context,
                List<String?> candidateData,
                List<dynamic> rejectedData,
                ) {
              final isHovering = candidateData.isNotEmpty;

              return AnimatedContainer(
                duration: const Duration(milliseconds: 200),
                width: 250,
                height: 150,
                alignment: Alignment.center,
                decoration: BoxDecoration(
                  color: isHovering
                      ? Colors.green.shade300
                      : Colors.grey.shade300,
                  borderRadius: BorderRadius.circular(20),
                  border: Border.all(
                    color: Colors.black26,
                    width: 2,
                  ),
                ),
                child: Text(
                  isHovering
                      ? 'Release to drop'
                      : 'Dropped: $droppedItem',
                  style: const TextStyle(
                    fontSize: 20,
                    fontWeight: FontWeight.bold,
                  ),
                ),
              );
            },
          ),
        ],
      ),
    );
  }

  Widget _buildDragBox({
    required Color color,
    required String label,
  }) {
    return Container(
      width: 100,
      height: 100,
      alignment: Alignment.center,
      decoration: BoxDecoration(
        color: color,
        borderRadius: BorderRadius.circular(16),
      ),
      child: Text(
        label,
        style: const TextStyle(
          color: Colors.black,
          fontWeight: FontWeight.bold,
        ),
      ),
    );
  }
}
