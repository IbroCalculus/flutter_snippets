A widget that shows a small dropdown menu when you tap it, usually used in the AppBar or as a trailing icon.
It can be used anywhere, not only in the AppBar.

----------------------------------------------------------

import 'package:flutter/material.dart';

class DemoPage extends StatelessWidget {
  const DemoPage({super.key});

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text('Popup Menu Example'),
        actions: [
          IconButton(onPressed: (){}, icon: const Icon(Icons.person)),
          const PopUpMenuButtonWidget(),                                  // USED HERE AS WELL
        ],
      ),
      body: const SafeArea(
        child: Center(child: PopUpMenuButtonWidget()),                    // USED HERE AS WELL
      ),
    );
  }
}

class PopUpMenuButtonWidget extends StatelessWidget {
  const PopUpMenuButtonWidget({super.key});

  @override
  Widget build(BuildContext context) {
    return PopupMenuButton<String>(
      onSelected: (value) {
        // Handle selection
        ScaffoldMessenger.of(context).showSnackBar(
          SnackBar(content: Text('You selected: $value')),
        );
      },

      itemBuilder: (context) => const [
        PopupMenuItem(
          value: 'profile',
          child: Text('Profile'),
        ),
        PopupMenuItem(
          value: 'settings',
          child: Text('Settings'),
        ),
        PopupMenuItem(
          value: 'logout',
          child: Text('Logout'),
        ),
      ],
    );
  }
}
