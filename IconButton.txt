Wrap an Icon widget with an IconButton widget        

        body: const Center(
          child: IconButton(
          iconSize: 100.0,
          color: Colors.blue,
          splashColor: Colors.lightGreen,
          highlightColor: Colors.red,

            icon: Icon(
              Icons.account_box_outlined,
              color: Colors.red,
            ),
            onPressed: null,
          ),
        ),
