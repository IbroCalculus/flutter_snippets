===== Creating/Accessing a collection (if already exists), document and pass fields ========

            ElevatedButton(
              onPressed: () async {

                final database = FirebaseFirestore.instance;				// ✅ Get Firestore instance
                final usersCollection = database.collection("users");			// ✅ Access a specific collection
                final userDoc = usersCollection.doc("1");				// ✅ Access a specific document inside "users" collection

NB: userDoc above defines a specific document. ignoring it and simply using usersCollection below will use autogenerated document ID
NB: for usersCollection = database.collection("users"); "users" is mostly the userCredential.user?.uid when logged in. Check Firebase auth - Email_Password 1 .txt on that.

                // ✅ Create or update a document

		await userDoc.set({					// HERE could simply be: await usersCollecition.set({... })
    			"username": "John",
    			"age": 30,
    			"city": "New York",
  		});
          },
       child: Text("Create Data"),

            ),

NB: .set      - replaces the whole document by default or creates the document if it doesn't exist.
    .add      - Creates a new document with an automatic ID.
    .update   - Updates specific fields ONLY.
    .delete   - Deletes a document.

---------- Enhanced of the above, include checking if the data was successfully uploaded or not -------------

ElevatedButton(
  onPressed: () async {
    final databaseReference = FirebaseFirestore.instance;
    final userReference = databaseReference.collection("users").doc("1");		// LIKEWISE, Can simply ignore .doc("1") so as to always get unique autogenerated document ID 

    // Create a Map with the data you want to set
    final userData = {
      "username": "John",
      "age": 45,
      "address": "New York",
    };

    // Use the set method and handle the response
    userReference.set(userData).then((_) {
      // If the set operation completes successfully
      print("Data created successfully!");
    }).catchError((error) {
      // If an error occurs during the set operation
      print("Error creating data: $error");
    });
  },
  child: Text("Create Data"),
),

  List<dynamic> notes = [];

  Future<void> getNotes() async {
    try {
      QuerySnapshot snapshot = await usersCollection.get();

      List<Map<String, dynamic>> fetchedNotes =
      snapshot.docs.map((doc) {
        return {
          "id": doc.id,
          ...doc.data() as Map<String, dynamic>,
        };
      }).toList();

      print("NOTE: $notes");

      setState(() {
        notes = fetchedNotes;
      });

    } on FirebaseException catch (e) {
      debugPrint("Firebase Error: ${e.message}");
    } catch (e) {
      debugPrint("Error: $e");
    }
  }
