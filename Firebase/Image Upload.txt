Future<String> _uploadImageToFirebase(User user, File imageFile) async {
  final storageRef = FirebaseStorage.instance
      .ref()
      .child('user_images')
      .child('${user.uid}.jpg');

  await storageRef.putFile(imageFile);
  return await storageRef.getDownloadURL();
}

---------------------- Example 2 -------------------------------

Two buttons, Pick Image and Upload Image. Use the image_picker to pick and then firebase storage


  File? _pickedImageFile;


  void _pickImage() async {
    final ImagePicker picker = ImagePicker();
    final XFile? pickedImage = await picker.pickImage(source: ImageSource.gallery, imageQuality: 50, maxWidth: 150);			// ImageSource.camera
    if (pickedImage == null) {
      return;
    }
    setState(() {
      _pickedImageFile = File(pickedImage.path);
    });

    ScaffoldMessenger.of(context).showSnackBar(
      const SnackBar(content: Text('Imaged picked successfully!')),
    );
  }


  Future<String> _uploadImage(File image) async {
    final storageRef = FirebaseStorage.instance.ref().child('gallery_images').child("uploaded_image");
    await storageRef.putFile(image);
    return await storageRef.getDownloadURL();
  }
