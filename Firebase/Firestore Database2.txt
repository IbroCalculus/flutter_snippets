  final _formKey = GlobalKey<FormState>();

  final TextEditingController _titleController = TextEditingController();
  final TextEditingController _descriptionController = TextEditingController();

  bool _isLoading = false;


  Future<void> _saveNote() async {

    if (!_formKey.currentState!.validate()) return;

    setState(() => _isLoading = true);

    try {
      final title = _titleController.text.trim();
      final description = _descriptionController.text.trim();

      final user = FirebaseAuth.instance.currentUser;

      if (user == null) {
        throw Exception("User not logged in");
      }

      final uid = user.uid;

      final data = await FirebaseFirestore.instance
          .collection("users")
          .doc(uid)
          .collection("notes")
          .add({
        "title": title,
        "description": description,
        "createdAt": FieldValue.serverTimestamp(),
      });

      print("CollectionId: ${data.id}");    // Retrieves the autogenerated document ID

      if (!mounted) return;

      ScaffoldMessenger.of(context).showSnackBar(
        const SnackBar(content: Text("Note added successfully")),
      );

      Navigator.pop(context);
    } on FirebaseException catch (e) {
      if (mounted) {
        ScaffoldMessenger.of(context).showSnackBar(
          SnackBar(content: Text("Firestore error: ${e.message}")),
        );
      }
    } catch (e) {
      if (mounted) {
        ScaffoldMessenger.of(context).showSnackBar(
          SnackBar(content: Text("Error: $e")),
        );
      }
    } finally {
      if (mounted) setState(() => _isLoading = false);
    }
  }

  ------------------------- 2b. Reading from firestore (with FutureBuilder) -------------------------

final FirebaseAuth auth = FirebaseAuth.instance;

@override
Widget build(BuildContext context) {

  final user = auth.currentUser;

  if (user == null) {
    return const Scaffold(
      body: Center(
        child: Text("No authenticated user"),
      ),
    );
  }

  final uid = user.uid;

  final Future<QuerySnapshot<Map<String, dynamic>>> notesFuture =
      FirebaseFirestore.instance
          .collection("users")
          .doc(uid)
          .collection("notes")
          .orderBy("createdAt", descending: true)
          .get();

  return Scaffold(
    appBar: AppBar(
      title: const Text("Main Page"),
      actions: [
        IconButton(
          icon: const Icon(Icons.logout),
          onPressed: () => _confirmLogout(context),
        ),
      ],
    ),

    body: FutureBuilder<QuerySnapshot<Map<String, dynamic>>>(
      future: notesFuture,

      builder: (context, snapshot) {

        // Waiting for Firestore data
        if (snapshot.connectionState == ConnectionState.waiting) {
          return const Center(
            child: CircularProgressIndicator(),
          );
        }

        // Firestore error
        else if (snapshot.hasError) {
          return Center(
            child: Text("Error: ${snapshot.error}"),
          );
        }

        // No data found
        else if (!snapshot.hasData || snapshot.data!.docs.isEmpty) {
          return const Center(
            child: Text("No notes found"),
          );
        }

        // Data retrieved successfully
        else if (snapshot.hasData) {

          final docs = snapshot.data!.docs;

          return ListView.builder(
            padding: const EdgeInsets.all(16),
            itemCount: docs.length,

            itemBuilder: (context, index) {

              final note = docs[index].data();

              final title = note["title"] ?? "";
              final description = note["description"] ?? "";

              return Card(
                margin: const EdgeInsets.only(bottom: 12),

                child: ListTile(
                  title: Text(title),
                  subtitle: Text(description),
                ),
              );
            },
          );
        }

        // Fallback state
        else {
          return const Center(
            child: Text("Something went wrong"),
          );
        }
      },
    ),

    floatingActionButton: FloatingActionButton(
      child: const Icon(Icons.add),

      onPressed: () {
        Navigator.push(
          context,
          MaterialPageRoute(
            builder: (_) => const AddNote(),
          ),
        );
      },
    ),
  );
}


  ------------------------- 3a. Reading from firestore (with Stream. StreamBuilder is below at 3b.) -------------------------

final FirebaseAuth auth = FirebaseAuth.instance;

void retrieveNotes() {
  try {
    final user = FirebaseAuth.instance.currentUser;

    if (user == null) {
      print("No user logged in");
      return;
    }

    final uid = user.uid;

    final Stream<QuerySnapshot<Map<String, dynamic>>> notesStream =
        FirebaseFirestore.instance
            .collection("users")
            .doc(uid)
            .collection("notes")
            .snapshots();

    notesStream.listen((snapshot) {

      print("Retrieved QuerySnapshot: ${snapshot.docs}");
      print("Total notes: ${snapshot.docs.length}");

      if (snapshot.docs.isEmpty) {
        print("No notes found");
        return;
      }

      for (var doc in snapshot.docs) {
          print("DATA: ${doc.data()}");
          print("NOTE ID: ${doc.id}");
          print("NOTE TITLE: ${doc.data()["title"]}");
          print("NOTE DESCRIPTION: ${doc.data()["description"]}");
          print("------------------------");
      }

    });

  } catch (e) {
    print("Error retrieving notes: $e");
  }
}
