---------------- Example 1 (Basic Usage of showModalBottomSheet) ---------------------
NB: showModalBottomSheet return a Future.

Just placed within onPressed(){}

void showBottomSheet() async {

print("Bottom sheet opened");

await showModalBottomSheet(
  context: context,
  isScrollControlled: true,
  builder: (BuildContext context) {
    return SizedBox(
      height: 200,
      child: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          mainAxisSize: MainAxisSize.min,
          children: <Widget>[
            const Text('Modal BottomSheet'),
            ElevatedButton(
              child: const Text('Close BottomSheet'),
              onPressed: () => Navigator.pop(context),
            ),
          ],
        ),
      ),
    );
  },
);

print("Bottom sheet closed");

}



---------------- Example 2 (An example that uses StatefulBuilder) ---------------------
NB: The usage of two setState methods: 
    ie; setState(() {}) which is global for the page, and setBottomSheetState(() {}) which is local for the bottom sheet

  int count = 0;

  void showBottomSheet() async {

    print("Bottom sheet Opened");

    await showModalBottomSheet(
      context: context,
      isScrollControlled: true,
      builder: (context) {
        return StatefulBuilder(
          builder: (context, setBottomSheetState) {
            return Container(
              height: 200,
              child: Center(
                child: Column(
                  mainAxisSize: MainAxisSize.min,
                  children: [
                    Text("Count Value: $count"),

                    ElevatedButton(
                      onPressed: () {
                        setBottomSheetState(() {
                          count++;
                        });
                      },
                      child: Text("Increment Count"),
                    ),

                    ElevatedButton(
                      onPressed: () {
                        setState(() {});
                        Navigator.of(context).pop();
                      },
                      child: Text("Close"),
                    ),
                  ],
                ),
              ),
            );
          },
        );
      },
    );

    print("Bottom sheet closed");
    setState(() {});
  }
