AspectRatio is a widget that forces its child to keep a fixed width-to-height ratio.
Think of it like: “No matter the screen or parent size, keep this shape consistent.”

----------------------------------------

import 'package:flutter/material.dart';

class DemoPage extends StatelessWidget {
  const DemoPage({super.key});

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text('AspectRatio Example')),
      body: const SafeArea(child: Center(child: AspectRatioWidget())),
    );
  }
}

class AspectRatioWidget extends StatelessWidget {
  const AspectRatioWidget({super.key});

  @override
  Widget build(BuildContext context) {
    return Container(
      width: 400,
      child: AspectRatio(
        aspectRatio: 16 / 9, // width : height


        child: Container(
          width: 100,
          color: Colors.blue,
          child: const Center(
            child: Text(
              '16:9 Box',
              style: TextStyle(color: Colors.white, fontSize: 18),
            ),
          ),
        ),
      ),
    );
  }
}
